import React, { useState } from "react";
import { Navbar, Container, Nav, InputGroup, Col, Row, Button } from "react-bootstrap";

const InfoBar = (props) => {
  const { temp, onTempChanged } = props;
  const [tempChecked, setTempChecked] = useState(false);

  const handleClick = () => {
    onTempChanged(tempChecked);
  };

  return (
    <Navbar bg="dark" data-bs-theme="dark">
      <Container>
        <Nav className="me-auto" style={{ color: "white" }}>
          <InputGroup style={{ width: "20vh" }}>
            <Col>
              <Row>
                <Col>
                  <p>Temperatūra</p>
                </Col>
                <Col>
                  <InputGroup.Checkbox
                    aria-label="Temperatūra"
                    checked={tempChecked}
                    onChange={() => setTempChecked(!tempChecked)}
                  />
                </Col>
              </Row>
              <Button onClick={handleClick}>Filter</Button>
            </Col>
          </InputGroup>
        </Nav>
      </Container>
    </Navbar>
  );
};

export default InfoBar;
